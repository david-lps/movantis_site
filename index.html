import { Movantis } from "@movantis/node";
const movantis = new Movantis(process.env.MOVANTIS_KEY);

const transfer = await movantis.transfers.create({
  source:      { country: "US", currency: "USD", amount: 500.0 },
  destination: { country: "MX", currency: "MXN", method: "bank_account" },
  routing:     "optimal",
  settlement:  "hybrid",
});

console.log(transfer.rail); // "rtp_spei"# Create a US → MX payout via Movantis Switch
